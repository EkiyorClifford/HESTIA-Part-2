```php
// Add to Admin.php class:
public function get_filtered_properties($filter = 'all', $keyword = '') {
    $sql = "SELECT p.*, s.state_name, l.lga_name, pt.type_name, u.first_name, u.last_name, u.email
            FROM properties p 
            JOIN property_types pt ON p.property_type_id = pt.type_id
            JOIN states s ON p.state_id = s.state_id
            JOIN lgas l ON p.lga_id = l.lga_id
            JOIN users u ON p.user_id = u.id";
    
    $params = [];
    $conditions = [];
    
    // Add keyword search
    if (!empty($keyword)) {
        $conditions[] = "(p.title LIKE ? OR p.description LIKE ? OR s.state_name LIKE ? OR l.lga_name LIKE ? OR u.first_name LIKE ? OR u.last_name LIKE ?)";
        $term = "%$keyword%";
        $params = array_merge($params, [$term, $term, $term, $term, $term, $term]);
    }
    
    // Add status filter
    if ($filter !== 'all') {
        $conditions[] = "p.status = ?";
        $params[] = $filter;
    }
    
    if (!empty($conditions)) {
        $sql .= " WHERE " . implode(" AND ", $conditions);
    }
    
    $sql .= " ORDER BY p.created_at DESC";
    
    $stmt = $this->dbconn->prepare($sql);
    $stmt->execute($params);
    return $stmt->fetchAll(PDO::FETCH_ASSOC);
}

// Then in property_management.php:
$all_properties = $admin->get_filtered_properties($filter, $keyword);
```
